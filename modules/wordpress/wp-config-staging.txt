    $_SERVER['HTTP_HOST'] = 'staging.@SITE_DOMAIN_EXCLUDING_WWW';
}

/** ********************************************************************************************* */
/** WP-Config (Staging): SSL Settings *********************************************************** */
/** ********************************************************************************************* */

define( 'FORCE_SSL_LOGIN', true ); // ss = true

    $_SERVER['HTTPS'] = 'on';
}

/** ********************************************************************************************* */
/** WP-Config (Staging): Cookie Settings ******************************************************** */
